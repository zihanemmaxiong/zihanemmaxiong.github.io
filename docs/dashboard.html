Introduction

This dashboard explores when and what customers purchase on the Instacart platform, focusing on product preferences, ordering time patterns, and reordering behaviors.

instacart_s |>
  count(aisle,sort = TRUE) |>
  top_n(10,n) |>
  mutate(aisle=fct_reorder(aisle,n)) |>
  plot_ly(
    x=~n,
    y=~aisle,
    type="bar",
    orientation="h",
    color = ~aisle,
    colors = "viridis",
    showlegend=FALSE
  ) |>
  layout(
    title="Top 10 Most Purchased Product Aisles",
    xaxis=list(title="Orders Numbers"),
    yaxis=list(title="")
  )
instacart_s |>
  group_by(order_dow, order_hour_of_day) |>
  summarise(num_orders = n(), .groups = "drop") |>
  mutate(
    order_dow = factor(order_dow,
                       levels = 0:6,
                       labels = c("Monday", "Tuesday", "Wednesday", "Thursday", "Friday", "Saturday", "Sunday"))
  ) |>
  plot_ly(
    x = ~order_hour_of_day,
    y = ~num_orders,
    color = ~order_dow,
    colors = "viridis",
    type = "scatter",
    mode = "lines+markers"
  ) |>
  layout(
    title = "Order Frequency Hour x Day of Week",
    xaxis = list(title = "Hour of Day"),
    yaxis = list(title = "Number of Orders")
  )
instacart_s |>
  plot_ly(
    x=~department,

layout(
